FastServe 教程

安装使用

直接引入fastserve.js即可

快速上手

const FastServe = require('./fastserve');
FastServe.createServer().listen(80);

参数设置

const FastServe = require('./fastserve');

/*参数设置*/
FastServe.config.cache   = false; //取消模板缓存 默认为true
FastServe.config.maxage  = 0; //取消静态文件缓存 默认值86400
FastServe.config.static  = './res'; //静态资源目录 默认./static
FastServe.config.charset = 'utf-8'; //静态资源编码 默认值utf-8
FastServe.config.encode  = 'utf8'; //动态输出编码 默认值utf8
FastServe.config.err404  = 'err'; //404错误提示 默认值Not Found
FastServe.config.https   = false; //是否为https 默认值false

FastServe.createServer().listen(80);
//FastServe.createServer({...}).listen(80);

响应输出

const FastServe = require('./fastserve');

//响应get请求
FastServe.get('/hello', (req, res) => {
	res.html('hello world!');
});

FastServe.createServer().listen(80);
